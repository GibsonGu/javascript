id:"id", //id必须防止重复
title:"标题" , //显示在弹出层的标题
url:"" //要加载的内容地址
$('#***').skyLayer.Sclose(); //关闭弹出层

$(function(){
　$('#img').click(function(){
　　$(this).skyLayer({id:'img',title:'图片',url:"./photo.html"});
　})
　$('#music').click(function(){
　　$(this).skyLayer({id:'music',title:'音乐',url:"./music.html"});
　})
　$('#doc').click(function(){
　　$(this).skyLayer({id:'doc',title:'说明',url:"./doc.html"});
　})
})
